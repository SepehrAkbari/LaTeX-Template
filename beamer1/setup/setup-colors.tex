% DESIGN COLORS
\definecolor{accent}{HTML}{203F5F} % accent color
\definecolor{bgcolor}{HTML}{FCFCFF} % background color
\definecolor{bgcolorAlt}{HTML}{E6ECF5} % alternative background color
\definecolor{fgcolor}{HTML}{222244} % foreground/text color

% accent : Accent color that is used for the progress bar and \alert{text}.
% bgcolor : Main background color that is used as the background color of the slides.
% bgcolorAlt : Second/alternative background color that is used as the background color of the slide titles and as the background color for boxes.
% fgcolor : Foreground color that is used as the text color.

% Lake Forest College: #C21430, #B8A66E, ECF1FC


%
\setbeamercolor{normal text}{%
	fg=fgcolor,
	bg=bgcolor,
}
\setbeamercolor{alerted text}{%
	fg=accent,
}
\setbeamercolor{palette primary}{%
	use=normal text,
	fg=normal text.fg,
	bg=bgcolorAlt,%normal text.bg
}

\setbeamercolor{block title}{
	bg=bgcolorAlt,
}
\setbeamercolor{block body}{
	bg=bgcolorAlt,
}
\setbeamercolor{block title alerted}{%
	use={palette primary, alerted text},
	fg=palette primary.bg,
	bg=alerted text.fg
}
\setbeamercolor{block title example}{%
	use={block title, alerted text},
	bg=block title.bg,
	fg=alerted text.fg
}
